Jupytext needs to be installed "globally"
  sudo apt update
  sudo apt install -y pipx
  pipx ensurepath    # adds ~/.local/bin to local PATH
  # Open new shell or: source ~/.bashrc / ~/.zshrc

  pipx install jupytext
  which jupytext     # Should display something like ~/.local/bin/jupytext 
  jupytext --version
